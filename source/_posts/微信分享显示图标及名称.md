# 网页在微信中分享，分享图片及名称的显示
<!-- more -->
```js
网页在微信中分享，分享图片及名称的显示
//微信分享图标显示
  getWXicon(){
    const script = document.createElement("script")
    script.text = `
       const storage = window.localStorage
      const title = storage.getItem("title")|| ""
      const describeText = storage.getItem("describeText")|| ""
    $("div.nav__main a.template__nav:eq(0)").addClass("on");
	function isWeiXin(){
        var ua = window.navigator.userAgent.toLowerCase();
        if(ua.match(/MicroMessenger/i) == 'micromessenger'){
            return true;
        }else{
            return false;
        }
    }
    if (isWeiXin()) {
        $.ajax({
            type:'GET',
            url: "/gwss?url=" + encodeURIComponent(location.href.split('#')[0]),
            success: function(data) {
                wx.config({
                    debug: false,
                    appId: data.appId,
                    timestamp: data.timestamp,
                    nonceStr: data.nonceStr,
                    signature: data.signature,
                    jsApiList: [
                        'checkJsApi',
                        'onMenuShareTimeline',
                        'onMenuShareAppMessage',
                        'onMenuShareQQ',
                        'onMenuShareWeibo',
                    ]
                });
                wx.ready(function () {
                    var shareData = {
                        title: title,
                        desc: describeText,
                        link: location.href.split('#')[0],
                        imgUrl: window.location.origin + "/static/images/cs_logo.jpg"
                    };
                    wx.onMenuShareAppMessage(shareData);
                    wx.onMenuShareTimeline(shareData);
                    wx.onMenuShareQQ(shareData);
                    wx.onMenuShareWeibo(shareData);
                });
            }
        })
    }

      `
    document.body.appendChild(script)
  }
```
